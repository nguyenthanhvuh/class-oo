class Point {
  private int x; private int y;
  public Point (int x, int y) {
      this.x = x; this.y = y;
  }
  @Override public boolean equals (Object o) {
     if (!(o instanceof Point)) { return false; }
     Point p = (Point) o;     
     return p.x == x && p.y == y;
  }
  @Override public int hashCode() { // See Question 3  }
}
class CounterPoint extends Point {
  private static int count = 0;   // point counter; not threadsafe
  public CounterPoint (int x, int y)  { super(x, y); count++;}
  public int getCount() { return count; }
}
